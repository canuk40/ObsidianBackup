┌──────────────────────────────────────────────────────────────────────────────┐
│ USAGE EXAMPLE                                                                │
└──────────────────────────────────────────────────────────────────────────────┘

  // 1. Create provider
  val provider = BoxCloudProvider(context, keystoreManager, logger, "account")

  // 2. Authenticate (OAuth2)
  val intent = provider.oauth2Provider.getAuthorizationIntent()
  startActivityForResult(intent, REQUEST_CODE)

  // 3. Handle callback
  val code = intent.data?.getQueryParameter("code")
  provider.oauth2Provider.exchangeCodeForToken(code, "account")

  // 4. Upload backup
  provider.uploadSnapshot(snapshotId, cloudFiles, metadata)

  // 5. Observe progress
  provider.observeProgress().collect { progress ->
      when (progress) {
          is Uploading -> updateUI(progress.bytesTransferred)
          is Completed -> showSuccess()
          is Failed -> showError(progress.error)
      }
  }
